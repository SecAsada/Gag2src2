local Players = game:GetService("Players")

local LocalPlayer = Players.LocalPlayer

_G.HideTrees = false
_G.HideFruits = false

-- Weak table cache for original transparency values
if not _G.PartCache then
	_G.PartCache = setmetatable({}, { __mode = "k" })
end

-- Helper function to find the dominant color of the plant
local function getPlantColor(plant)
	for _, desc in ipairs(plant:GetDescendants()) do
		if desc:IsA("BasePart")
			and desc.Name ~= "Fruits"
			and desc.Name ~= "Base"
			and not desc:IsA("TrussPart")
			and tonumber(desc.Name) == nil
		then
			if desc.Transparency < 1 then
				return desc.BrickColor
			end
		end
	end
	return BrickColor.new("Parsley green") -- fallback
end

-- Helper function to handle visibility + recolor
local function updateVisibility(instance, isHidden, plantColor)
	local function cacheIfNeeded(obj)
		if _G.PartCache[obj] == nil then
			_G.PartCache[obj] = obj.Transparency
		end
	end

	local function applyStateToPart(part)
		-- Only cache TRUE native transparency when we're NOT hiding.
		-- Prevents caching 1.0 as "native" if parts spawn while hidden.
		if not isHidden then
			cacheIfNeeded(part)
		end

		local nativeTrans = _G.PartCache[part]
		if nativeTrans == nil then
			nativeTrans = 0
		end

		local desiredTrans = isHidden and 1 or nativeTrans
		if part.Transparency ~= desiredTrans then
			part.Transparency = desiredTrans
		end

		-- Recolor grey structural parts back to plant color (only when visible)
		if not isHidden then
			if (part:IsA("TrussPart")
				or part.Name == "Base"
				or part.Name == "CollisionBlock"
				or tonumber(part.Name) ~= nil)
			then
				local cName = part.BrickColor.Name
				if cName == "Medium stone grey" or cName == "Light stone grey" then
					part.BrickColor = plantColor
				end
			end
		end
	end

	local function applyStateToDecal(decal)
		-- For decals/textures, caching is always safe (no recolor)
		cacheIfNeeded(decal)
		local nativeTrans = _G.PartCache[decal] or 0
		local desiredTrans = isHidden and 1 or nativeTrans
		if decal.Transparency ~= desiredTrans then
			decal.Transparency = desiredTrans
		end
	end

	-- Apply to instance itself
	if instance:IsA("BasePart") then
		applyStateToPart(instance)
	elseif instance:IsA("Decal") or instance:IsA("Texture") then
		applyStateToDecal(instance)
	end

	-- Apply to descendants
	for _, desc in ipairs(instance:GetDescendants()) do
		if desc:IsA("BasePart") then
			applyStateToPart(desc)
		elseif desc:IsA("Decal") or desc:IsA("Texture") then
			applyStateToDecal(desc)
		end
	end
end

-- Refresh transparency updates on your current plot plants
local function updatePlotVisuals()
	local plotId = LocalPlayer:GetAttribute("PlotId")
	if not plotId then return end

	local gardens = Workspace:FindFirstChild("Gardens")
	if not gardens then return end

	local plot = gardens:FindFirstChild("Plot" .. tostring(plotId))
	if not plot then return end

	local plants = plot:FindFirstChild("Plants")
	if not plants then return end

	for _, plant in ipairs(plants:GetChildren()) do
		local pColor = getPlantColor(plant)

		for _, child in ipairs(plant:GetChildren()) do
			if child.Name == "Fruits" then
				updateVisibility(child, _G.HideFruits, pColor)
			else
				updateVisibility(child, _G.HideTrees, pColor)
			end
		end
	end
end

-- Automatic monitoring loop
task.spawn(function()
	while task.wait(0.05) do
		updatePlotVisuals()
	end
end)
