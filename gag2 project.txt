local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
    Name = "Sec Asada : Grow a Garden 2",
    LoadingTitle = "SEC ASADA",
    LoadingSubtitle = "@Secre_Dev",
    ConfigurationSaving = { Enabled = false },
    Discord = { Enabled = false },
    KeySystem = false,
})

local function hideTreeOrFruit(model)
    for _, part in ipairs(model:GetDescendants()) do
        if part:IsA("BasePart") then
            part.Transparency = 1
            part.CanCollide = false
        end
    end
end

local function scanAndHide()
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Model") and (obj.Name:match("Tree") or obj.Name:match("Fruit")) then
            hideTreeOrFruit(obj)
        end
    end
end

-- Run once
scanAndHide()

-- Watch for new models
workspace.DescendantAdded:Connect(function(child)
    if child:IsA("Model") and (child.Name:match("Tree") or child.Name:match("Fruit")) then
        hideTreeOrFruit(child)
    end
end)

MainTab:CreateToggle({
    Name = "Hide Trees & Fruits",

    Callback = function(value)
        isEnabled = value
