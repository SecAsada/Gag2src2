local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")

-- Global control states
_G.HideTrees = _G.HideTrees or false
_G.HideFruits = _G.HideFruits or false

-- Cache original values (weak keys so destroyed parts get cleaned up)
if not _G.PartCache then
    _G.PartCache = setmetatable({}, { __mode = "k" })
end

-- Cache format:
-- _G.PartCache[obj] = { T = originalTransparency, LTM = originalLocalTransparencyModifier }

local function rememberOriginal(obj)
    if _G.PartCache[obj] ~= nil then return end

    local t = 0
    local ltm = 0

    if obj:IsA("BasePart") then
        t = obj.Transparency
        ltm = obj.LocalTransparencyModifier
    elseif obj:IsA("Decal") or obj:IsA("Texture") then
        t = obj.Transparency
        ltm = nil
    else
        return
    end

    _G.PartCache[obj] = { T = t, LTM = ltm }
end

local function restoreOriginal(obj)
    local cached = _G.PartCache[obj]
    if not cached then
        -- If we never cached it (ex: spawned while hidden), assume visible defaults
        if obj:IsA("BasePart") then
            obj.Transparency = 0
            obj.LocalTransparencyModifier = 0
        elseif obj:IsA("Decal") or obj:IsA("Texture") then
            obj.Transparency = 0
        end
        return
    end

    if obj:IsA("BasePart") then
        obj.Transparency = cached.T or 0
        obj.LocalTransparencyModifier = cached.LTM or 0
    elseif obj:IsA("Decal") or obj:IsA("Texture") then
        obj.Transparency = cached.T or 0
    end
end

local function setHidden(obj, hidden)
    if obj:IsA("BasePart") then
        if hidden then
            obj.Transparency = 1
            obj.LocalTransparencyModifier = 1
        else
            restoreOriginal(obj)
        end
    elseif obj:IsA("Decal") or obj:IsA("Texture") then
        if hidden then
            obj.Transparency = 1
        else
            restoreOriginal(obj)
        end
    end
end

local function getPlantColor(plant)
    for _, desc in ipairs(plant:GetDescendants()) do
        if desc:IsA("BasePart")
            and desc.Name ~= "Fruits"
            and desc.Name ~= "Base"
            and not desc:IsA("TrussPart")
            and tonumber(desc.Name) == nil
        then
            -- IMPORTANT: use the part's real transparency (excluding LocalTransparencyModifier weirdness)
            if desc.Transparency < 1 then
                return desc.BrickColor
            end
        end
    end
    return BrickColor.new("Parsley green")
end

local function recolorGreyStructuralParts(root, plantColor)
    for _, part in ipairs(root:GetDescendants()) do
        if part:IsA("BasePart") then
            if (part:IsA("TrussPart") or part.Name == "Base" or part.Name == "CollisionBlock" or tonumber(part.Name) ~= nil) then
                local bn = part.BrickColor.Name
                if bn == "Medium stone grey" or bn == "Light stone grey" then
                    part.BrickColor = plantColor
                end
            end
        end
    end
end

local function updatePlotVisuals()
    local plotId = game:GetService("Players").LocalPlayer:GetAttribute("PlotId")
    if not plotId then return end

    local gardens = workspace:FindFirstChild("Gardens")
    if not gardens then return end

    local plot = gardens:FindFirstChild("Plot" .. tostring(plotId))
    if not plot then return end

    local plantsFolder = plot:FindFirstChild("Plants")
    if not plantsFolder then return end

    for _, plant in ipairs(plantsFolder:GetChildren()) do
        local pColor = getPlantColor(plant)

        -- Cache originals ONLY while visible-mode.
        -- This prevents "hidden-time caching" from permanently saving Transparency=1.
        if not _G.HideTrees then
            for _, obj in ipairs(plant:GetDescendants()) do
                if obj:IsA("BasePart") or obj:IsA("Decal") or obj:IsA("Texture") then
                    rememberOriginal(obj)
                end
            end
        end
        if not _G.HideFruits then
            local fruits = plant:FindFirstChild("Fruits")
            if fruits then
                for _, obj in ipairs(fruits:GetDescendants()) do
                    if obj:IsA("BasePart") or obj:IsA("Decal") or obj:IsA("Texture") then
                        rememberOriginal(obj)
                    end
                end
            end
        end

        -- Apply hidden/visible state
        for _, obj in ipairs(plant:GetDescendants()) do
            if obj:IsA("BasePart") or obj:IsA("Decal") or obj:IsA("Texture") then
                local isFruit = (obj:FindFirstAncestor("Fruits") ~= nil)
                if isFruit then
                    setHidden(obj, _G.HideFruits)
                else
                    setHidden(obj, _G.HideTrees)
                end
            end
        end

        -- Only recolor when trees are visible (optional, matches your intent)
        if not _G.HideTrees then
            recolorGreyStructuralParts(plant, pColor)
        end
    end
end

-- Loop (you can keep your loop rate; 0.05 is fine but a bit heavy)
task.spawn(function()
    while task.wait(0.1) do
        updatePlotVisuals()
    end
end)
