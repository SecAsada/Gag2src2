-- MAIN TABS & VARIABLES

local MainTab = Window:CreateTab("Main Features", nil)

local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local LocalPlayer = Players.LocalPlayer

-- Global control states matching your logic
_G.HideTrees = false
_G.HideFruits = false

-- Weak table to safely cache the original native transparency of parts
if not _G.PartCache then
    _G.PartCache = setmetatable({}, {__mode = "k"})
end

-- Helper function to find the dominant color of the plant
local function getPlantColor(plant)
    for _, desc in ipairs(plant:GetDescendants()) do
        if desc:IsA("BasePart") and desc.Name ~= "Fruits" and desc.Name ~= "Base" and not desc:IsA("TrussPart") and tonumber(desc.Name) == nil then
            if desc.Transparency < 1 then
                return desc.BrickColor
            end
        end
    end
    return BrickColor.new("Parsley green") -- Fallback normal color
end

-- Helper function to handle visibility and colors seamlessly
local function updateVisibility(instance, isHidden, plantColor)
    local function applyState(part)
    -- Only learn/cache the TRUE native transparency when we are NOT hiding.
    -- This prevents caching 1.0 as the "native" value when parts spawn while hidden.
    if not isHidden then
        if _G.PartCache[part] == nil then
            _G.PartCache[part] = part.Transparency
        end
    end

    local nativeTrans = _G.PartCache[part]
    if nativeTrans == nil then
        -- Fallback: if we never cached it (because we were hidden), assume visible.
        nativeTrans = 0
    end

    local desiredTrans = isHidden and 1 or nativeTrans
    if part.Transparency ~= desiredTrans then
        part.Transparency = desiredTrans
    end

    -- recolor logic stays the same
    if part:IsA("BasePart") and not isHidden then
        if (part:IsA("TrussPart") or part.Name == "Base" or part.Name == "CollisionBlock" or tonumber(part.Name) ~= nil) then
            if part.BrickColor.Name == "Medium stone grey" or part.BrickColor.Name == "Light stone grey" then
                part.BrickColor = plantColor
            end
        end
    end
end
        
        local nativeTrans = _G.PartCache[part]
        
        -- If hiding, force 1. If visible, RESTORE to its normal native transparency
        local desiredTrans = isHidden and 1 or nativeTrans
        
        if part.Transparency ~= desiredTrans then
            part.Transparency = desiredTrans
        end
        
        -- 2. Recolor any ugly grey structural parts to the plant's normal color
        if part:IsA("BasePart") and not isHidden then
            if (part:IsA("TrussPart") or part.Name == "Base" or part.Name == "CollisionBlock" or tonumber(part.Name) ~= nil) then
                if part.BrickColor.Name == "Medium stone grey" or part.BrickColor.Name == "Light stone grey" then
                    part.BrickColor = plantColor
                end
            end
        end
    end

    if instance:IsA("BasePart") then
        applyState(instance)
    elseif instance:IsA("Decal") or instance:IsA("Texture") then
        if _G.PartCache[instance] == nil then _G.PartCache[instance] = instance.Transparency end
        local dTrans = isHidden and 1 or _G.PartCache[instance]
        if instance.Transparency ~= dTrans then instance.Transparency = dTrans end
    end
    
    for _, desc in ipairs(instance:GetDescendants()) do
        if desc:IsA("BasePart") then
            applyState(desc)
        elseif desc:IsA("Decal") or desc:IsA("Texture") then
            if _G.PartCache[desc] == nil then _G.PartCache[desc] = desc.Transparency end
            local dTrans = isHidden and 1 or _G.PartCache[desc]
            if desc.Transparency ~= dTrans then desc.Transparency = dTrans end
        end
    end
end

-- Helper function to refresh transparency updates on your current plot plants
local function updatePlotVisuals()
    local plotId = LocalPlayer:GetAttribute("PlotId")
    if not plotId then return end
    
    local gardens = Workspace:FindFirstChild("Gardens")
    if not gardens then return end
    
    local plot = gardens:FindFirstChild("Plot" .. tostring(plotId))
    if not plot then return end
    
    local plants = plot:FindFirstChild("Plants")
    if not plants then return end

    for _, plant in ipairs(plants:GetChildren()) do
        local pColor = getPlantColor(plant)
        
        for _, child in ipairs(plant:GetChildren()) do
            if child.Name == "Fruits" then
                -- ---------- Hide/Show Fruits ----------
                updateVisibility(child, _G.HideFruits, pColor)
            else
                -- ---------- Hide/Show Trees ----------
                updateVisibility(child, _G.HideTrees, pColor)
            end
        end
    end
end

-- Automatic dynamic monitoring for when new plants grow or load into your plot
task.spawn(function()
    while task.wait(0.05) do
        updatePlotVisuals()
    end
end)

-- ============================================================

    Flag = "HideTreesPlotFlag",
    Callback = function(Value)
        _G.HideTrees = Value
