-- ============================================================
-- HIDE TREES & FRUITS (NO UI) - KEYBINDS VERSION
-- ============================================================
local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local LocalPlayer = Players.LocalPlayer
local UserInputService = game:GetService("UserInputService")

-- Toggle states
local hideTrees = false
local hideFruits = false

-- Function to update visibility on your plot's plants
local function updatePlot()
    local plotId = LocalPlayer:GetAttribute("PlotId")
    if not plotId then return end

    local gardens = Workspace:FindFirstChild("Gardens")
    if not gardens then return end

    local plot = gardens:FindFirstChild("Plot" .. tostring(plotId))
    if not plot then return end

    local plants = plot:FindFirstChild("Plants")
    if not plants then return end

    for _, plant in ipairs(plants:GetChildren()) do
        -- ---- Hide/Show Tree Trunk & Leaves (all non-Fruit parts) ----
        for _, child in ipairs(plant:GetChildren()) do
            if child.Name ~= "Fruits" then
                -- If it's a BasePart directly
                if child:IsA("BasePart") then
                    child.Transparency = hideTrees and 1 or 0
                end
                -- If it's a Model or Folder, go through descendants
                for _, part in ipairs(child:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.Transparency = hideTrees and 1 or 0
                    end
                end
            end
        end

        -- ---- Hide/Show Fruits ----
        local fruitsFolder = plant:FindFirstChild("Fruits")
        if fruitsFolder then
            for _, fruit in ipairs(fruitsFolder:GetChildren()) do
                for _, part in ipairs(fruit:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.Transparency = hideFruits and 1 or 0
                    end
                end
            end
        end
    end
end

-- Automatic refresh when new plants appear (check every second)
task.spawn(function()
    while task.wait(1) do
        updatePlot()
    end
end)

-- Keybind: press G to toggle Trees
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == Enum.KeyCode.G then
        hideTrees = not hideTrees
        updatePlot()
        print("Trees hidden:", hideTrees)
    elseif input.KeyCode == Enum.KeyCode.H then
        hideFruits = not hideFruits
        updatePlot()
        print("Fruits hidden:", hideFruits)
    end
end)

print("Script loaded! Press G to toggle trees, H to toggle fruits.")
