-- ================= SUPER SCANNER FOR MAIL REMOTE =================
local MailRemote = nil
local FoundRemotes = {}

local function scanForRemote()
    FoundRemotes = {}
    local candidates = {}

    -- List of services to scan (including all common hiding spots)
    local servicesToScan = {
        ReplicatedStorage,
        game:GetService("Workspace"),
        game:GetService("Players"),
        game:GetService("ReplicatedFirst"),
        game:GetService("StarterGui"),
        game:GetService("StarterPack"),
        game:GetService("Lighting"),
        game:GetService("SoundService"),
        game:GetService("Chat"),
        game:GetService("ScriptContext"),
        -- Some devs put remotes in the local player itself
        LocalPlayer,
        LocalPlayer:FindFirstChild("PlayerGui"),
        LocalPlayer:FindFirstChild("Backpack"),
        LocalPlayer:FindFirstChild("Character"),
    }

    for _, service in ipairs(servicesToScan) do
        if service then
            for _, obj in pairs(service:GetDescendants()) do
                if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
                    local nameLower = string.lower(obj.Name)
                    -- Keywords: mail, send, gift, item, transfer
                    if string.find(nameLower, "mail") or 
                       string.find(nameLower, "send") or 
                       string.find(nameLower, "gift") or 
                       string.find(nameLower, "transfer") or
                       string.find(nameLower, "item") then
                        table.insert(candidates, obj)
                    end
                end
            end
        end
    end

    -- If multiple candidates, pick the most likely (prefer "SendMail" or "MailRemote")
    if #candidates > 0 then
        for _, c in ipairs(candidates) do
            if c.Name == "SendMail" or c.Name == "MailRemote" then
                FoundRemotes = candidates
                return c
            end
        end
        FoundRemotes = candidates
        return candidates[1] -- fallback to first found
    end

    return nil
end

-- Initial scan
MailRemote = scanForRemote()

-- Remote Firing Helper Function (handles both RemoteEvent and RemoteFunction)
local function sendSingleTicket(recipient, itemName, count, message)
    if not MailRemote then 
        MailRemote = scanForRemote() 
    end
    if not MailRemote then return false, "No remote" end

    local success, err = pcall(function()
        if MailRemote:IsA("RemoteEvent") then
            MailRemote:FireServer(recipient, itemName, count, message)
        elseif MailRemote:IsA("RemoteFunction") then
            MailRemote:InvokeServer(recipient, itemName, count, message)
        end
    end)
    if not success then
        return false, err
    end
    return true
end

-- 9. Command Button: SEND MAIL BATCHES
MainTab:CreateButton({
   Name = "📨 SEND MAIL BATCHES",
   Callback = function()
       if IsSending then
           Rayfield:Notify({
              Title = "Blocked",
              Content = "A mailing sequence is already running!",
              Duration = 3,
              Image = 4483300124,
           })
           return
       end

       -- Data Validations
       if not MailRemote then
           MailRemote = scanForRemote()
           if not MailRemote then
               Rayfield:Notify({
                  Title = "Error",
                  Content = "Cannot locate mailing remote! Use 'Rescan Remote'.",
                  Duration = 5,
                  Image = 4483300124,
               })
               return
           end
       end

       if TargetRecipient == "" then
           Rayfield:Notify({
              Title = "Missing Recipient",
              Content = "Please type a recipient username first.",
              Duration = 4,
              Image = 4483300124,
           })
           return
       end

       if SelectedItemName == "" then
           Rayfield:Notify({
              Title = "Missing Item",
              Content = "Please select an item or seed from the list.",
              Duration = 4,
              Image = 4483300124,
           })
           return
       end

       -- Execute separate thread
       task.spawn(function()
           IsSending = true
           local remainingItems = TotalAmountToSend

           Rayfield:Notify({
              Title = "Process Begun",
              Content = string.format("Preparing to send %d %s to %s...", TotalAmountToSend, SelectedItemName, TargetRecipient),
              Duration = 4,
              Image = 4483300124,
           })

           while remainingItems > 0 do
               local ticketAmount = math.min(remainingItems, 20)
               local sent, err = sendSingleTicket(TargetRecipient, SelectedItemName, ticketAmount, CustomNote)

               if sent then
                   remainingItems = remainingItems - ticketAmount
                   Rayfield:Notify({
                      Title = "✅ Ticket Sent",
                      Content = string.format("Mailed %d %s. %d remaining.", ticketAmount, SelectedItemName, remainingItems),
                      Duration = 1.5,
                      Image = 4483300124,
                   })
               else
                   Rayfield:Notify({
                      Title = "❌ Mailing Stopped",
                      Content = "Failed to communicate: " .. (err or "unknown error"),
                      Duration = 5,
                      Image = 4483300124,
                   })
                   break
               end

               if remainingItems > 0 then
                   task.wait(DelayBetweenTickets)
               end
           end

           if remainingItems == 0 then
               Rayfield:Notify({
                  Title = "🎉 Complete!",
                  Content = string.format("All %d %s have been sent!", TotalAmountToSend, SelectedItemName),
                  Duration = 5,
                  Image = 4483300124,
               })
           end

           IsSending = false
       end)
   end,
})
