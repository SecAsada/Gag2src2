-- ================= NEW DEEP-SCANNER FOR REMOTE =================
local MailRemote = nil

local function scanForRemote()
    -- Look for common exact names first
    local exact = ReplicatedStorage:FindFirstChild("SendMail", true) or ReplicatedStorage:FindFirstChild("MailRemote", true)
    if exact and exact:IsA("RemoteEvent") then
        return exact
    end
    
    -- Deep scan all game services where a developer might hide the remote
    local servicesToScan = {ReplicatedStorage, game:GetService("JointsService"), game:GetService("NetworkClient")}
    for _, service in ipairs(servicesToScan) do
        if service then
            for _, v in pairs(service:GetDescendants()) do
                if v:IsA("RemoteEvent") then
                    local nameLower = string.lower(v.Name)
                    -- Check if "mail" or "send" is in the name
                    if string.find(nameLower, "mail") or (string.find(nameLower, "send") and string.find(nameLower, "item")) then
                        return v
                    end
                end
            end
        end
    end
    return nil
end

MailRemote = scanForRemote()

-- Current State Configurations
local SelectedCategory = "Seeds"
local AvailableItems = Seeds
local SelectedItemName = ""
local TargetRecipient = ""
local TotalAmountToSend = 20
local CustomNote = ""
local IsSending = false
local DelayBetweenTickets = 1.5 

-- Load Rayfield Library
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "GAG2 Global Mailer",

-- UI TAB
local MainTab = Window:CreateTab("Mailer Core", 4483345998)

-- Warn player if remote was NOT found instantly on load

    Rayfield:Notify({
       Title = "Remote Warning",
       Content = "Mailing system not found! Try reopening or contact support.",
       Duration = 8,
       Image = 4483300124,

-- 1. Input: Recipient Username

   PlaceholderText = "Type recipient...",

   end,
})

-- 2. Dropdown: Choose Category (Seeds or Gears)
local ItemDropdown 

MainTab:CreateDropdown({
   Name = "Item Category",
   Options = {"Seeds", "Gears"},
   CurrentOption = "Seeds",
   MultipleOptions = false,
   Callback = function(Option)
      SelectedCategory = Option[1] or Option
      if SelectedCategory == "Seeds" then
          AvailableItems = Seeds
      else
          AvailableItems = Gears
      end
      
      if ItemDropdown then
          ItemDropdown:Refresh(AvailableItems, true)
      end
   end,
})

-- 3. Dropdown: Select Item from Category
ItemDropdown = MainTab:CreateDropdown({
   Name = "Select Item",
   Options = AvailableItems,
   CurrentOption = "",
   MultipleOptions = false,
   Callback = function(Option)
       SelectedItemName = Option[1] or Option
   end,
})

-- 4. Search Filter
MainTab:CreateInput({
   Name = "Search Items",
   PlaceholderText = "Type here to filter items...",
   RemoveTextAfterFocusLost = false,
   Callback = function(Text)
      local query = string.lower(Text)
      if query == "" then
          ItemDropdown:Refresh(AvailableItems, true)
      else
          local filtered = {}
          for _, item in ipairs(AvailableItems) do
              if string.find(string.lower(item), query) then
                  table.insert(filtered, item)
              end
          end
          ItemDropdown:Refresh(filtered, true)
      end
   end,
})

-- 5. Input: Quantity

   PlaceholderText = "Default: 20",
   RemoveTextAfterFocusLost = false,
   Callback = function(Text)
      local num = tonumber(Text)
      if num and num > 0 then
          TotalAmountToSend = math.floor(num)
      end
   end,
})

-- 6. Input: Custom Note / Message
MainTab:CreateInput({

   PlaceholderText = "Type message to include...",

   Callback = function(Text)
      CustomNote = Text
   end,
})

-- Remote Firing Helper Function
local function sendSingleTicket(recipient, itemName, count, message)
    -- Run deep-scan check again just in case the remote loaded late
    if not MailRemote then 
        MailRemote = scanForRemote() 
    end
    
    if not MailRemote then return false end
    
    local success, err = pcall(function()
        MailRemote:FireServer(recipient, itemName, count, message)
    end)
    return success
end

-- 7. Command Button: SEND MAIL
MainTab:CreateButton({
   Name = "SEND MAIL BATCHES",
   Callback = function()
       if IsSending then
           Rayfield:Notify({
              Title = "Blocked",
              Content = "A mailing sequence is already running!",
              Duration = 3,
              Image = 4483300124,
           })
           return
       end

       -- Data Validations
       if not MailRemote then
           -- Try one last scan
           MailRemote = scanForRemote()
           if not MailRemote then
               Rayfield:Notify({
                  Title = "Error",
                  Content = "Still cannot locate mailing remote! Game may have blocked external mail edits.",
                  Duration = 5,
                  Image = 4483300124,
               })
               return
           end
       end
       
       if TargetRecipient == "" then
           Rayfield:Notify({
              Title = "Missing Recipient",
              Content = "Please type a recipient username first.",
              Duration = 4,
              Image = 4483300124,
           })
           return
       end

       if SelectedItemName == "" then
           Rayfield:Notify({
              Title = "Missing Item",
              Content = "Please select an item or seed from the list.",
              Duration = 4,
              Image = 4483300124,
           })
           return
       end

       -- Execute separate thread
       task.spawn(function()
           IsSending = true
           local remainingItems = TotalAmountToSend
           
           Rayfield:Notify({
              Title = "Process Begun",
              Content = string.format("Preparing to send %d %s to %s...", TotalAmountToSend, SelectedItemName, TargetRecipient),
              Duration = 4,
              Image = 4483300124,
           })
           
           while remainingItems > 0 do
               local ticketAmount = math.min(remainingItems, 20)
               local sent = sendSingleTicket(TargetRecipient, SelectedItemName, ticketAmount, CustomNote)
               
               if sent then
                   remainingItems = remainingItems - ticketAmount
                   Rayfield:Notify({
                      Title = "Ticket Sent Successfully",
                      Content = string.format("Mailed %d %s. %d remaining.", ticketAmount, SelectedItemName, remainingItems),
                      Duration = 1.5,
                      Image = 4483300124,
                   })
               else
                   Rayfield:Notify({
                      Title = "Mailing Stopped",
                      Content = "Failed to communicate with Server Remote.",
                      Duration = 4,
                      Image = 4483300124,
                   })
                   break
               end
               
               if remainingItems > 0 then
                   task.wait(DelayBetweenTickets) 
               end
           end
           
           if remainingItems == 0 then
               Rayfield:Notify({
                  Title = "Complete!",
                  Content = string.format("All %d %s have been sent!", TotalAmountToSend, SelectedItemName),
                  Duration = 5,
                  Image = 4483300124,
               })
           end
           
           IsSending = false
       end)
   end,
})
